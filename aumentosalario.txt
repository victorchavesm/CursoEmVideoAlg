nome: caractere
salario, anos, aumento: real
inicio

   escreva ("Digite seu nome: ")
   leia (nome)
   escreva ("Digite seu salário: R$ ")
   leia (salario)
   escreva ("Há quantos anos você trabalha nessa empresa? ")
   leia (anos)
   
   se (anos) <= 3 entao
      aumento <- salario + (salario * (3/100))
      escreva ("Seu novo salário será R$ " , aumento)
   senao
        se (anos > 3) e (anos <= 10) entao
           aumento <- salario + (salario * (12.5/100))
           escreva ("Seu novo salário será R$ " , aumento)
        senao
             se (anos) > 10 entao
                aumento <- salario + (salario * (20/100))
                escreva ("Seu novo salário será R$ " , aumento)
             fimse
        fimse
   fimse
